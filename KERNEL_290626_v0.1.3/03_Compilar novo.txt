# 1. Montagem da Imagem de Disco Virtual

rm -f disco_total.img
dd if=/dev/zero of=disco_total.img bs=1M count=100
dd if=boot.bin of=disco_total.img bs=512 count=1 conv=notrunc
dd if=kernel.bin of=disco_total.img bs=512 seek=1 conv=notrunc

dd if=disco_sata.img of=disco_total.img seek=2048 conv=notrunc


# 2. Inicialização Segura no QEMU com Geometria Controlada

qemu-system-x86_64 -m 2G \
    -device ich9-ahci,id=ahci \
    -drive file=disco_total.img,if=none,id=disk,format=raw \
    -device ide-hd,bus=ahci.0,drive=disk,cyls=1024,heads=16,secs=255 \
    -d int,cpu_reset \
    -no-reboot \
    -no-shutdown
   
   
mcopy -o -i disco_sata.img explorer.elf ::/
